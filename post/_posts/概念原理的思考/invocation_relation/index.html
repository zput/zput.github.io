package main

import "fmt"

type YY struct {
	y string
}

func main() {
	var xx interface{}
	xx = nil

	if jj, ok := xx.(*YY); ok {
		fmt.Println("success assert", jj)
	} else {
		fmt.Println("failure assert")
	}

	var jj = xx.(*YY)

	if jj == nil {
		fmt.Println(jj)
	}
}
